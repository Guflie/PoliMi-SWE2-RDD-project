
\subsubsection*{Account creation Service}
\begin{table}[H]
    \centering
    \renewcommand{\arraystretch}{1.5}     
    \begin{tabular}{|l|p{0.8\textwidth}|}
        \hline
        Name & \multicolumn{1}{c|}{\textbf{Send Account Creation confirmation}} \\ 
        \hline
        Signature & send\_acc\_creat(name, surname, gender, birthdate, emailAddr, psw): void \\
        \hline
        Description & Sends to the user via email the link to confirm account creation, after executing the appropiate checks. \\
        \hline
        Parameters &
            \begin{minipage}{0.8\textwidth}
            \smallskip
            \begin{itemize}
                \item name: string, user's first name
                \item surname: string, user's last name
                \item gender: string, user's gender
                \item birthdate: date, user's date of birth
                \item emailAddr: string, user's email address
                \item psw: string, user's chosen password
            \end{itemize}
            \smallskip
            \end{minipage}  
        \\
        \hline
        Returned & \\
        \hline
        Exceptions & The email is already registered for an account (already\_used\_email\_error) \\
        \hline
    \end{tabular}
\end{table}

            \end{minipage}  
        \\
        \hline
        Returned & Returns a success status (acc\_creation\_succ) \\
        \hline
        Exceptions & The token validation failed or timed out (link\_expired\_error) \\
        \hline
    \end{tabular}
\end{table}
